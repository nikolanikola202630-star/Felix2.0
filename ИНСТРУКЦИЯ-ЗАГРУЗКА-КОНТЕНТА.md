### Способ 3: Через API (Программно)

**Создайте скрипт для загрузки:**

```javascript
// scripts/upload-courses.js
const { createClient } = require('@supabase/supabase-js');
const fs = require('fs');

const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_KEY
);

async function uploadCourses() {
  // Загрузить курсы из JSON файла
  const courses = JSON.parse(fs.readFileSync('./data/courses.json', 'utf8'));
  
  for (const course of courses) {
    const { data, error } = await supabase
      .from('courses')
      .insert(course);
    
    if (error) {
      console.error(`❌ Error uploading ${course.title}:`, error);
    } else {
      console.log(`✅ Uploaded: ${course.title}`);
    }
  }
}

uploadCourses();
```

**Запустите:**

```bash
node scripts/upload-courses.js
```

## 🔄 Обновление контента

### Обновить курс:

```sql
UPDATE courses
SET 
  title = 'Новое название',
  description = 'Новое описание',
  updated_at = NOW()
WHERE id = 'course-id';
```

### Добавить урок:

```sql
UPDATE courses
SET themes = jsonb_set(
  themes,
  '{0,lessons}',
  themes->'0'->'lessons' || '[{"id": 5, "title": "Новый урок", "type": "text", "duration": "10 мин", "content": "Содержимое"}]'::jsonb
)
WHERE id = 'course-id';
```

### Удалить курс:

```sql
DELETE FROM courses WHERE id = 'course-id';
```
