## 🔧 Технические детали

### Голосовой ввод

```javascript
// Запись голоса
let mediaRecorder;
let audioChunks = [];

async function startRecording() {
  const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
  mediaRecorder = new MediaRecorder(stream);
  
  mediaRecorder.ondataavailable = (event) => {
    audioChunks.push(event.data);
  };
  
  mediaRecorder.onstop = async () => {
    const audioBlob = new Blob(audioChunks, { type: 'audio/webm' });
    const text = await transcribeAudio(audioBlob);
    document.getElementById('messageInput').value = text;
    audioChunks = [];
  };
  
  mediaRecorder.start();
}

function stopRecording() {
  mediaRecorder.stop();
}

async function transcribeAudio(audioBlob) {
  const formData = new FormData();
  formData.append('audio', audioBlob);
  
  const response = await fetch('/api/voice/transcribe', {
    method: 'POST',
    body: formData
  });
  
  const data = await response.json();
  return data.text;
}
```

### Память AI

```javascript
// Управление историей
class ChatHistory {
  constructor(userId) {
    this.userId = userId;
    this.maxMessages = 10;
    this.load();
  }
  
  load() {
    const stored = localStorage.getItem(`chat_history_${this.userId}`);
    this.messages = stored ? JSON.parse(stored) : [];
  }
  
  save() {
    localStorage.setItem(`chat_history_${this.userId}`, JSON.stringify(this.messages));
  }
  
  add(role, content) {
    this.messages.push({ role, content, timestamp: new Date().toISOString() });
    
    // Ограничение количества сообщений
    if (this.messages.length > this.maxMessages * 2) {
      this.messages = this.messages.slice(-this.maxMessages * 2);
    }
    
    this.save();
  }
  
  getContext() {
    return this.messages.slice(-this.maxMessages);
  }
  
  clear() {
    this.messages = [];
    this.save();
  }
}

// Использование
const history = new ChatHistory(userId);

async function sendMessage() {
  const message = document.getElementById('messageInput').value;
  
  // Добавить в историю
  history.add('user', message);
  
  // Отправить с контекстом
  const response = await fetch('/api/ai/chat', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      message,
      history: history.getContext()
    })
  });
  
  const data = await response.json();
  
  // Добавить ответ в историю
  history.add('assistant', data.response);
}
```
