```sql
-- lessons
CREATE TABLE lessons (
  id SERIAL PRIMARY KEY,
  course_id INTEGER REFERENCES courses(id),
  title VARCHAR(255) NOT NULL,
  description TEXT,
  content_type VARCHAR(50), -- 'video', 'text', 'test'
  content_url TEXT, -- YouTube/Vimeo URL
  content_text TEXT,
  duration INTEGER, -- секунды
  order_num INTEGER,
  is_free BOOLEAN DEFAULT false,
  created_at TIMESTAMP DEFAULT NOW()
);

-- lesson_progress
CREATE TABLE lesson_progress (
  id SERIAL PRIMARY KEY,
  user_id BIGINT REFERENCES users(id),
  lesson_id INTEGER REFERENCES lessons(id),
  completed BOOLEAN DEFAULT false,
  completed_at TIMESTAMP,
  watch_time INTEGER DEFAULT 0,
  UNIQUE(user_id, lesson_id)
);

-- purchases
CREATE TABLE purchases (
  id SERIAL PRIMARY KEY,
  user_id BIGINT REFERENCES users(id),
  course_id INTEGER REFERENCES courses(id),
  amount DECIMAL(10,2),
  currency VARCHAR(10) DEFAULT 'RUB',
  payment_method VARCHAR(50), -- 'telegram_stars', 'card'
  payment_id VARCHAR(255),
  status VARCHAR(50) DEFAULT 'pending',
  referrer_partner_id BIGINT,
  referrer_user_id BIGINT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- partner_payouts
CREATE TABLE partner_payouts (
  id SERIAL PRIMARY KEY,
  partner_id BIGINT REFERENCES partner_accounts(user_id),
  amount DECIMAL(10,2),
  method VARCHAR(50), -- 'card', 'crypto'
  details JSONB, -- реквизиты
  status VARCHAR(50) DEFAULT 'pending',
  requested_at TIMESTAMP DEFAULT NOW(),
  processed_at TIMESTAMP,
  processed_by BIGINT
);

-- bonus_transactions
CREATE TABLE bonus_transactions (
  id SERIAL PRIMARY KEY,
  user_id BIGINT REFERENCES users(id),
  amount DECIMAL(10,2),
  type VARCHAR(50), -- 'earned', 'spent'
  description TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```

Создать `api/payments.js`:

```javascript
const { db } = require('../lib/db');

module.exports = async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { course_id, user_id } = req.body;
  
  // Получить курс
  const course = await db.query(
    'SELECT * FROM courses WHERE id = $1',
    [course_id]
  );
  
  if (!course.rows[0]) {
    return res.status(404).json({ error: 'Course not found' });
  }

  const courseData = course.rows[0];
  
  // Создать инвойс для Telegram Stars
  const invoice = {
    title: courseData.title,
    description: courseData.description,
    payload: JSON.stringify({ course_id, user_id }),
    provider_token: '', // Пусто для Stars
    currency: 'XTR', // Telegram Stars
    prices: [{
      label: courseData.title,
      amount: Math.floor(courseData.price * 100) // В копейках Stars
    }]
  };

  return res.json({ invoice });
};
```

Создать `api/partner-stats.js`:

```javascript
const { db } = require('../lib/db');

module.exports = async function handler(req, res) {
  const userId = req.query.user_id;
  
  // Проверить, что пользователь - партнер
  const partner = await db.getPartnerAccount(userId);
  if (!partner) {
    return res.status(403).json({ error: 'Not a partner' });
  }
  
  // Статистика
  const stats = await db.query(`
    SELECT
      COUNT(DISTINCT rc.id) as total_clicks,
      COUNT(DISTINCT CASE WHEN u.id IS NOT NULL THEN u.id END) as registrations,
      COUNT(DISTINCT p.id) as purchases,
      COALESCE(SUM(p.amount * 0.2), 0) as total_earned,
      COALESCE(SUM(CASE 
        WHEN p.created_at >= NOW() - INTERVAL '30 days' 
        THEN p.amount * 0.2 
        ELSE 0 
      END), 0) as month_earned
    FROM referral_clicks rc
    LEFT JOIN users u ON u.referrer_partner_id = $1
    LEFT JOIN purchases p ON p.user_id = u.id AND p.status = 'completed'
    WHERE rc.partner_user_id = $1
  `, [userId]);
  
  return res.json({
    ...stats.rows[0],
    balance: partner.balance,
    referral_link: `https://t.me/AcademyBot?start=ref_partner${userId}`
  });
};
```

**6.2. API для админки**

Создать `api/admin.js`:

```javascript
const { db } = require('../lib/db');

module.exports = async function handler(req, res) {
  // Проверка прав администратора
  const userId = req.headers['x-user-id'];
  const isAdmin = await checkAdmin(userId);
  
  if (!isAdmin) {
    return res.status(403).json({ error: 'Forbidden' });
  }

  const { action } = req.query;

  switch (action) {
    case 'create_course':
      return await createCourse(req, res);
    case 'create_lesson':
      return await createLesson(req, res);
    case 'list_partners':
      return await listPartners(req, res);
    case 'approve_payout':
      return await approvePayout(req, res);
    case 'stats':
      return await getStats(req, res);
    default:
      return res.status(400).json({ error: 'Unknown action' });
  }
};

async function createCourse(req, res) {
  const { title, description, price, category, image } = req.body;
  
  const result = await db.query(`
    INSERT INTO courses (title, description, price, category, image_url)
    VALUES ($1, $2, $3, $4, $5)
    RETURNING *
  `, [title, description, price, category, image]);
  
  return res.json(result.rows[0]);
}

async function createLesson(req, res) {
  const { course_id, title, content_type, content_url, is_free } = req.body;
  
  const result = await db.query(`
    INSERT INTO lessons (course_id, title, content_type, content_url, is_free)
    VALUES ($1, $2, $3, $4, $5)
    RETURNING *
  `, [course_id, title, content_type, content_url, is_free]);
  
  return res.json(result.rows[0]);
}

async function listPartners(req, res) {
  const result = await db.query(`
    SELECT 
      pa.*,
      u.first_name,
      u.username,
      COUNT(DISTINCT rc.id) as total_clicks,
      COUNT(DISTINCT p.id) as total_purchases,
      COALESCE(SUM(p.amount * 0.2), 0) as total_earned
    FROM partner_accounts pa
    JOIN users u ON u.id = pa.user_id
    LEFT JOIN referral_clicks rc ON rc.partner_user_id = pa.user_id
    LEFT JOIN purchases p ON p.referrer_partner_id = pa.user_id
    GROUP BY pa.user_id, u.first_name, u.username
    ORDER BY total_earned DESC
  `);
  
  return res.json(result.rows);
}

async function approvePayout(req, res) {
  const { payout_id, admin_id } = req.body;
  
  await db.query(`
    UPDATE partner_payouts
    SET status = 'approved', processed_at = NOW(), processed_by = $2
    WHERE id = $1
  `, [payout_id, admin_id]);
  
  // Списать с баланса партнера
  const payout = await db.query(
    'SELECT * FROM partner_payouts WHERE id = $1',
    [payout_id]
  );
  
  await db.query(`
    UPDATE partner_accounts
    SET balance = balance - $1
    WHERE user_id = $2
  `, [payout.rows[0].amount, payout.rows[0].partner_id]);
  
  return res.json({ success: true });
}

async function getStats(req, res) {
  const stats = await db.query(`
    SELECT
      (SELECT COUNT(*) FROM users) as total_users,
      (SELECT COUNT(*) FROM purchases WHERE status = 'completed') as total_purchases,
      (SELECT COALESCE(SUM(amount), 0) FROM purchases WHERE status = 'completed') as total_revenue,
      (SELECT COUNT(*) FROM partner_accounts WHERE is_active = true) as active_partners
  `);
  
  return res.json(stats.rows[0]);
}
```
