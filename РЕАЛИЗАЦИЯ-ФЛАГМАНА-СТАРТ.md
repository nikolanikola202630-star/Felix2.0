```javascript
// miniapp/js/app.js
async function apiRequest(url, data) {
  const initData = window.Telegram.WebApp.initData;
  
  const response = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Telegram-WebApp-Init-Data': initData
    },
    body: JSON.stringify(data)
  });
  
  if (response.status === 429) {
    const data = await response.json();
    tg.showAlert(`Слишком много запросов. Попробуйте через ${data.retryAfter} сек.`);
    return null;
  }
  
  return response.json();
}
```

**Код для `miniapp/course.html`:**
```javascript
async function buyCourse(courseId) {
  try {
    // Показать загрузку
    tg.MainButton.showProgress();
    
    // Создать invoice
    const response = await fetch(`${CONFIG.API_URL}/payments/create-invoice`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-Telegram-WebApp-Init-Data': tg.initData
      },
      body: JSON.stringify({
        course_id: courseId,
        user_id: tg.initDataUnsafe.user.id
      })
    });
    
    const data = await response.json();
    
    if (!data.success) {
      throw new Error(data.error);
    }
    
    // Скрыть загрузку
    tg.MainButton.hideProgress();
    
    // Открыть invoice
    tg.openInvoice(data.invoice_link, (status) => {
      if (status === 'paid') {
        tg.showAlert('✅ Оплата успешна! Курс доступен.');
        window.location.reload();
      } else if (status === 'cancelled') {
        tg.showAlert('Оплата отменена');
      } else if (status === 'failed') {
        tg.showAlert('Ошибка оплаты. Попробуйте снова.');
      }
    });
    
  } catch (error) {
    console.error('Buy course error:', error);
    tg.MainButton.hideProgress();
    tg.showAlert('Ошибка: ' + error.message);
  }
}
```
