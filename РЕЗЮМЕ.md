## 💡 Ключевые модули

### lib/cache.js
```javascript
// Кэширование с Vercel KV
import { cache } from './lib/cache.js';

// Получить с кэшем
const settings = await cache.get(
  `userSettings:${userId}`,
  () => db.getUserSettings(userId),
  3600 // 1 час
);

// Инвалидация
await cache.invalidate(`userSettings:${userId}`);
```

### lib/monitoring.js
```javascript
// Мониторинг с Sentry
import { captureError, logger } from './lib/monitoring.js';

try {
  // код
} catch (error) {
  captureError(error, { userId, action: 'command' });
}

logger.info('User action', { userId, command });
```

### lib/ai-rate-limit.js
```javascript
// AI Rate Limiting
import { checkAILimit, incrementAIUsage } from './lib/ai-rate-limit.js';

// Проверка лимита
const canUse = await checkAILimit(userId);
if (!canUse) {
  return 'Превышен лимит AI запросов';
}

// Инкремент после успеха
await incrementAIUsage(userId, tokens);
```
