## 🎯 Что должно управляться через админку

### 1. Управление курсами

**Основная информация:**
- ✅ Название курса
- ✅ Описание
- ✅ Категория (psychology, self-development, business, it, trading)
- ✅ Уровень (beginner, intermediate, advanced, all)
- ✅ Цена в рублях (автоконвертация в Stars)
- ✅ Изображение курса (URL)
- ✅ Длительность в часах
- ✅ Рейтинг
- ✅ Количество студентов

**Преподаватель:**
- ✅ Имя
- ✅ Должность/звание
- ✅ Аватар (эмодзи)

**Статус:**
- ✅ Опубликован / Черновик
- ✅ Видимость (публичный / приватный)
- ✅ Дата создания / обновления

### 2. Управление темами курса

**Для каждой темы:**
- ✅ Название темы
- ✅ Порядковый номер
- ✅ Описание
- ✅ PDF файл (опционально)
- ✅ Добавить / Удалить / Переместить

### 3. Управление уроками

**Для каждого урока:**
- ✅ Название урока
- ✅ Описание
- ✅ Длительность (минуты)
- ✅ Видео URL (YouTube, Vimeo, прямая ссылка)
- ✅ Бесплатный / Платный
- ✅ Порядковый номер
- ✅ Материалы (PDF, ссылки)
- ✅ Тест/Задание (опционально)

### 4. Ценообразование

**Настройки цен:**
- ✅ Цена в рублях
- ✅ Автоматический расчет в Stars (1 Star = 2₽)
- ✅ Скидки (процент, фиксированная сумма)
- ✅ Промокоды

## 🔧 Функции админ панели

### 1. Быстрые действия

```javascript
// Дублировать курс
function duplicateCourse(courseId) {
  const course = getCourse(courseId);
  const newCourse = {
    ...course,
    id: generateId(),
    title: course.title + ' (копия)',
    students: 0,
    created_at: new Date()
  };
  saveCourse(newCourse);
}

// Массовое изменение цен
function bulkUpdatePrices(courseIds, discount) {
  courseIds.forEach(id => {
    const course = getCourse(id);
    course.price = Math.floor(course.price * (1 - discount / 100));
    saveCourse(course);
  });
}

// Экспорт курса
function exportCourse(courseId) {
  const course = getCourse(courseId);
  const json = JSON.stringify(course, null, 2);
  downloadFile(`course-${courseId}.json`, json);
}

// Импорт курса
function importCourse(file) {
  const reader = new FileReader();
  reader.onload = (e) => {
    const course = JSON.parse(e.target.result);
    course.id = generateId();
    saveCourse(course);
  };
  reader.readAsText(file);
}
```

### 2. Валидация данных

```javascript
function validateCourse(course) {
  const errors = [];
  
  if (!course.title || course.title.length < 3) {
    errors.push('Название должно быть не менее 3 символов');
  }
  
  if (!course.description || course.description.length < 10) {
    errors.push('Описание должно быть не менее 10 символов');
  }
  
  if (course.price < 0 || course.price > 100000) {
    errors.push('Цена должна быть от 0 до 100,000₽');
  }
  
  if (!course.themes || course.themes.length === 0) {
    errors.push('Добавьте хотя бы одну тему');
  }
  
  course.themes.forEach((theme, i) => {
    if (!theme.lessons || theme.lessons.length === 0) {
      errors.push(`Тема ${i + 1}: добавьте хотя бы один урок`);
    }
  });
  
  return errors;
}
```

### 3. Автосохранение

```javascript
let autoSaveTimer = null;

function enableAutoSave() {
  document.querySelectorAll('input, textarea, select').forEach(el => {
    el.addEventListener('input', () => {
      clearTimeout(autoSaveTimer);
      autoSaveTimer = setTimeout(() => {
        saveDraft();
        showNotification('Черновик сохранен', 'success');
      }, 2000);
    });
  });
}

function saveDraft() {
  const course = collectFormData();
  localStorage.setItem('course_draft', JSON.stringify(course));
}

function loadDraft() {
  const draft = localStorage.getItem('course_draft');
  if (draft) {
    const course = JSON.parse(draft);
    fillForm(course);
    showNotification('Черновик восстановлен', 'info');
  }
}
```

function previewCourse(course) {
  // Открыть в новом окне
  const previewWindow = window.open('', '_blank');
  
  // Сгенерировать HTML предпросмотра
  const html = generateCoursePreview(course);
  
  previewWindow.document.write(html);
  previewWindow.document.close();
}

function generateCoursePreview(course) {
  return `
    <!DOCTYPE html>
    <html>
    <head>
      <title>${course.title} - Предпросмотр</title>
      <style>/* стили курса */</style>

      <div class="course-header">
        <h1>${course.title}</h1>

        <div class="price">${course.price} ₽</div>
      </div>
      <!-- остальной контент -->

    </html>
  `;
}

function exportReport(type, format) {
  const data = getReportData(type);
  
  if (format === 'csv') {
    const csv = convertToCSV(data);
    downloadFile(`report-${type}.csv`, csv);
  } else if (format === 'excel') {

    downloadFile(`report-${type}.xlsx`, excel);
